# Configurações do Supabase
# Copie este arquivo para .env e substitua pelos seus valores reais

# URL do projeto Supabase (encontre em Settings > API)
SUPABASE_URL=https://seu-projeto.supabase.co

# Chave de serviço (encontre em Settings > API > service_role key)
SUPABASE_SERVICE_KEY=sua-service-role-key-aqui

# Chave pública (encontre em Settings > API > anon key)
SUPABASE_KEY=sua-anon-key-aqui

# Nome do bucket de storage (crie em Storage)
SUPABASE_STORAGE_BUCKET=roupas

# Configurações do banco PostgreSQL
# Encontre em Settings > Database > Connection string
SUPABASE_DB_HOST=db.seu-projeto.supabase.co
SUPABASE_DB_NAME=postgres
SUPABASE_DB_USER=postgres
SUPABASE_DB_PASSWORD=sua-senha-do-banco
SUPABASE_DB_PORT=5432
